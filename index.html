# ClinicalTriageEnv v5 — Full Integration Demo # Real graders + Llama 3 inference + LLM reward shaping import requests BASE = "https://samrudh-nux-ClinicalTriageEnv.hf.space" # 1. Check all modules loaded health = requests.get(f"{BASE}/health").json() print(health["modules"]) # shows which .py files loaded # 2. Reset real RL environment (uses environment.py + graders.py) r = requests.post(f"{BASE}/reset", json={"task_id": "triage_hard"}).json() session_id = r["session_id"] # 3. Submit action → real graders + LLM evaluation step = requests.post(f"{BASE}/step", json={ "session_id": session_id, "reasoning": "ESI-1: FAST+, LKW 1h45m, warfarin unknown INR", "use_llm_eval": True, "action": { "esi_level": 1, "rationale": "Acute ischaemic stroke — activate stroke alert", "recommended_immediate_interventions": [ "Stroke alert activation", "CT head without contrast", "IV access × 2", "INR stat" ] } }).json() print(f"Rule reward: {step['rule_reward']:.3f}") print(f"LLM eval: {step['llm_evaluation']}") print(f"Final reward: {step['reward']:.3f}") print(f"Formula: {step['reward_formula']}") # 4. Run Llama 3 inference (requires HF_TOKEN) llama = requests.post(f"{BASE}/inference/run", json={"task_id": "triage_hard", "use_cot": True}).json() print(f"Llama reward: {llama['result']['reward']:.3f}") # 5. Get oracle ("what would a doctor do?") oracle = requests.post(f"{BASE}/rl/oracle", json={ "state": {"task_type": "triage", "patient": {"vitals": {"spo2": 89, "systolic_bp": 75}, "chief_complaint": "unresponsive, collapse"}} }).json() print(f"Oracle ESI: {oracle['oracle_action']['esi_level']}")
